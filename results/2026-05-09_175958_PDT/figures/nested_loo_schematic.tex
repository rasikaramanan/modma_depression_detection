\begin{figure}[t]
\centering
\begin{tikzpicture}[
  font=\footnotesize,
  box/.style       = {draw, rounded corners=2pt, align=center,
                      inner sep=3pt, minimum height=5mm, fill=white},
  pbox/.style      = {box, fill=blue!4, text width=58mm},
  outerbox/.style  = {box, fill=gray!12, text width=66mm, font=\footnotesize\bfseries},
  hobox/.style     = {box, fill=red!10,  text width=28mm},
  trbox/.style     = {box, fill=blue!10, text width=33mm},
  predbox/.style   = {box, fill=green!10, text width=78mm},
  inhdr/.style     = {box, fill=blue!8, text width=66mm,
                      font=\footnotesize\bfseries},
  gridhdr/.style   = {box, fill=violet!10, text width=72mm,
                      font=\footnotesize\bfseries},
  selbox/.style    = {box, fill=yellow!18, text width=66mm,
                      font=\footnotesize\bfseries},
  ihbox/.style     = {box, fill=red!10, text width=24mm, font=\footnotesize},
  itbox/.style     = {box, fill=blue!10, text width=28mm, font=\footnotesize},
  sidenote/.style  = {font=\scriptsize\itshape, text=gray!60!black,
                      align=center, inner sep=1pt},
  >={Stealth[length=1.4mm]},
]

% --- Outer fold layer ---
\node[outerbox] (outer) {Outer fold $k \in \{1, \dots, 52\}$};

\node[hobox, below left=3mm and 2mm of outer.south, anchor=north east]
  (ho) {Held-out:\\ subject $k$};
\node[trbox, below right=3mm and 2mm of outer.south, anchor=north west]
  (tr) {Training:\\ 51 subjects};

\draw[->] (outer.south) -- (ho.north);
\draw[->] (outer.south) -- (tr.north);

% --- Grid search header (middle layer of the three nested loops) ---
\node[gridhdr, below=8mm of outer.south, yshift=-7mm] (gridhdr)
  {Grid search over ($g \in \{1, \dots, 32\}$) HP value combinations of $C$, $\epsilon$, $\gamma$};

\draw[->] (tr.south) -- (gridhdr.north);

% --- Inner LOO header (nested inside the grid-search layer) ---
\node[inhdr, below=3mm of gridhdr.south] (inhdr)
  {Inner LOSO over 51 outer-training subjects ($j \in \{1, \dots, 51\}$) with HP value combination $g$} ;

\draw[->] (gridhdr.south) -- (inhdr.north);

% --- Inner-fold split (1 inner held-out + 50 inner training) ---
\node[ihbox, below left=3mm and 2mm of inhdr.south, anchor=north east]
  (ho_inner) {Inner held-out:\\ subject $j$};
\node[itbox, below right=3mm and 2mm of inhdr.south, anchor=north west]
  (tr_inner) {Inner training:\\ 50 subjects};

\draw[->] (inhdr.south) -- (ho_inner.north);
\draw[->] (inhdr.south) -- (tr_inner.north);

% --- Pipeline (vertical, fits on 50 inner-training subjects) ---
\node[pbox, below=8mm of inhdr.south, yshift=-4mm] (impute)
  {Impute (median) and scale features};
\node[pbox, below=1.5mm of impute] (en)
  {Elastic Net feature selection};
\node[pbox, below=1.5mm of en] (svr)
  {SVR (RBF kernel) predicts inner held-out $j$};

\draw[->] (tr_inner.south) to[out=-90, in=90] (impute.north);
\draw[->] (impute) -- (en);
\draw[->] (en)     -- (svr);

% Background highlight — inner-CV ring only (gridbox dashed square removed
% per request; the grid-search layer is now conveyed by the violet header
% alone).
\begin{scope}[on background layer]
  \node[draw, rounded corners=3pt, fill=blue!2,
        fit=(inhdr) (ho_inner) (tr_inner)
            (impute) (en) (svr),
        inner sep=3pt] (innerbox) {};
\end{scope}

% --- Select best hyperparameters + refit on 51 outer-training subjects ---
\node[selbox, below=4mm of innerbox.south] (select)
  {Select $C^*, \epsilon^*, \gamma^*$ that minimize RMSE and refit on 51 training subjects};
\draw[->] (innerbox.south) -- (select.north);

% --- Predict outer held-out k ---
\node[predbox, below=4mm of select.south] (predict)
  {Predict held-out subject $k$};
\draw[->] (select.south) -- (predict.north);

\end{tikzpicture}
\caption{Nested leave-one-subject-out evaluation pipeline. HP abbreviates hyperparameter.}
\label{fig:nested-loo}
\end{figure}
